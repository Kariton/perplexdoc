You could end up with more than 100 images. At least 20 percent are recommended." $sizestep }} {{ else if (lt $sizestep 15) }} {{ warnf "The size difference of %d percent for your srcset is small. At least 20 percent are recommended." $sizestep }} {{ end }} {{ $m := (math.Sqrt (add 1 (div site.Params.imaging.sizestep 100.0))) }} {{ $n := (math.Ceil (div (math.Log $ratio) (math.Log $m))) }} {{ $mo := (math.Pow $ratio (div 1.0 $n)) }} {{ $size := $min }} {{ range (seq 1 (int $n)) }} {{ $size = (mul $size $mo) }} {{ $sizes = $sizes | append (int (math.Round $size)) }} {{ end }} {{ return $sizes }}
